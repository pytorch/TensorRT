        "\n# 3D Rotary Position Embedding (RoPE) + Attention compiled with Torch-TensorRT\n\n3D RoPE is the positional encoding used in video generation transformers such\nas CogVideoX, Wan, and HunyuanVideo.  Unlike 1D RoPE (used in language models)\nwhich encodes a single sequence index, 3D RoPE independently encodes three\naxes \u2014 temporal (T), height (H), and width (W) \u2014 and assigns each axis a\ndedicated slice of the per-head frequency vector:\n\n    head-dim slots  0 ..  d//3-1  \u2192 temporal  frequencies\n    head-dim slots  d//3.. 2d//3-1 \u2192 height    frequencies\n    head-dim slots  2d//3.. d//2-1 \u2192 width     frequencies\n\nThe rotation is expressed with complex arithmetic:\n\n    xq_rotated = view_as_real(view_as_complex(xq) * freqs_cis)\n\nPyTorch complex ops (view_as_complex, complex mul) are not natively supported\nby TensorRT.  Torch-TensorRT's ``complex_graph_detection`` lowering pass\nintercepts them before partitioning and rewrites the subgraph to equivalent\nreal arithmetic \u2014 splitting the last dimension into (..., 2) real/imag pairs\nand computing (ac-bd, ad+bc) manually \u2014 so the TRT engine only sees standard\nfloat32 ops and the caller never needs to change anything.\n\nThis example:\n  1. Defines a 3D-RoPE frequency precomputation helper (complex64 output).\n  2. Defines a VideoAttentionBlock: linear QKV projection \u2192 3D RoPE \u2192 SDPA.\n  3. Runs a PyTorch baseline forward pass.\n  4. Exports with torch.export.export() and dynamic T/H/W dimensions.\n  5. Compiles to TensorRT via torch_tensorrt.dynamo.compile().\n  6. Verifies numerical accuracy (cosine similarity on the output tensor).\n  7. (Optional) benchmarks latency of both backends.\n\n## Usage\n# Quick correctness check (static shapes)\npython examples/dynamo/torch_export_3d_rope.py\n\n# Dynamic T/H/W shapes\npython examples/dynamo/torch_export_3d_rope.py --dynamic\n\n# Larger config + benchmark\npython examples/dynamo/torch_export_3d_rope.py --heads 16 --head-dim 96 --t 8 --h 16 --w 16 --benchmark\n"

        "import argparse\nimport timeit\n\nimport torch\nimport torch.nn as nn\nimport torch.nn.functional as F\nimport torch_tensorrt\nfrom torch.export import Dim\n\nDEVICE = torch.device(\"cuda:0\")\n\n\n# ---------------------------------------------------------------------------\n# Frequency precomputation\n# ---------------------------------------------------------------------------\n\n\ndef precompute_freqs_3d(\n    head_dim: int,\n    t: int,\n    h: int,\n    w: int,\n    theta: float = 10000.0,\n) -> torch.Tensor:\n    \"\"\"Pre-compute 3D RoPE unit-complex frequency tensor.\n\n    Returns a complex64 tensor of shape (t, h, w, head_dim // 2) where the\n    last dimension is split evenly across the three spatial axes.\n\n    Args:\n        head_dim: Channels per attention head (must be even, head_dim//2\n                  must be divisible by 3).\n        t: Number of temporal frames.\n        h: Spatial height in patches.\n        w: Spatial width in patches.\n        theta: Base for the geometric frequency progression.\n    \"\"\"\n    half = head_dim // 2\n    d_t = half // 3\n    d_h = half // 3\n    d_w = half - d_t - d_h  # absorbs any remainder from integer division\n\n    def _axis_freqs(d: int, n: int) -> torch.Tensor:\n        \"\"\"1-D complex exponentials, shape (n, d).\"\"\"\n        inv_freq = 1.0 / (theta ** (torch.arange(0, d * 2, 2).float() / (d * 2)))\n        positions = torch.arange(n, dtype=torch.float32)\n        angles = torch.outer(positions, inv_freq)\n        return torch.polar(torch.ones_like(angles), angles)  # complex64\n\n    freqs_t = _axis_freqs(d_t, t)[:, None, None, :].expand(t, h, w, d_t)\n    freqs_h = _axis_freqs(d_h, h)[None, :, None, :].expand(t, h, w, d_h)\n    freqs_w = _axis_freqs(d_w, w)[None, None, :, :].expand(t, h, w, d_w)\n\n    # Concatenate along last dim \u2192 (t, h, w, half), complex64\n    return torch.cat([freqs_t, freqs_h, freqs_w], dim=-1).contiguous()\n\n\n# ---------------------------------------------------------------------------\n# Model\n# ---------------------------------------------------------------------------\n\n\nclass VideoAttentionBlock(nn.Module):\n    \"\"\"Single attention block for video latents with 3D RoPE.\n\n    Inputs\n    ------\n    x             : (B, T, H, W, C)  float32  video patch features\n    freqs_cis_real: (T, H, W, C // n_heads)  float32\n        The RoPE frequency tensor pre-flattened from complex64 via\n        ``view_as_real(...).flatten(-2)``.  The module reconstructs the\n        complex form internally with ``view_as_complex``.\n\n        Passing frequencies as a plain real-valued input avoids exposing a\n        complex tensor at the model boundary (TRT inputs must be real).\n\n    Output\n    ------\n    (B, T, H, W, C)  float32\n    \"\"\"\n\n    def __init__(self, channels: int = 512, n_heads: int = 8) -> None:\n        super().__init__()\n        assert channels % n_heads == 0\n        self.n_heads = n_heads\n        self.head_dim = channels // n_heads\n        self.norm = nn.LayerNorm(channels)\n        self.qkv = nn.Linear(channels, 3 * channels, bias=False)\n        self.proj = nn.Linear(channels, channels, bias=False)\n\n    def _apply_rope(self, x: torch.Tensor, freqs_cis: torch.Tensor) -> torch.Tensor:\n        \"\"\"Apply 3D RoPE to a single Q or K tensor.\n\n        The complex multiply ``xc * freqs_cis`` is what Torch-TensorRT rewrites\n        to real arithmetic via the complex_graph_detection lowering pass.\n\n        Args:\n            x        : (B, T, H, W, n_heads, head_dim)  float32\n            freqs_cis: (T, H, W, head_dim // 2)  complex64\n        Returns:\n            Rotated tensor, same shape as ``x``, float32.\n        \"\"\"\n        B, T, H, W, Nh, D = x.shape\n        # Interpret consecutive pairs of head-dim channels as complex numbers.\n        xc = torch.view_as_complex(x.reshape(B, T, H, W, Nh, D // 2, 2))\n        # freqs_cis broadcast over batch (dim 0) and head (dim 4).\n        freqs = freqs_cis[None, :, :, :, None, :]  # (1, T, H, W, 1, D//2)\n        return torch.view_as_real(xc * freqs).flatten(-2)  # (B,T,H,W,Nh,D)\n\n    def forward(\n        self,\n        x: torch.Tensor,\n        freqs_cis_real: torch.Tensor,\n    ) -> torch.Tensor:\n        B, T, H, W, C = x.shape\n        Nh, D = self.n_heads, self.head_dim\n\n        h = self.norm(x)\n        qkv = self.qkv(h).reshape(B, T, H, W, 3, Nh, D)\n        q, k, v = qkv.unbind(dim=4)  # each (B, T, H, W, Nh, D)\n\n        # Recover complex frequencies from the real-valued input.\n        # freqs_cis_real: (T, H, W, D) \u2192 reshape to (T, H, W, D//2, 2) \u2192 complex\n        freqs_cis = torch.view_as_complex(freqs_cis_real.reshape(T, H, W, D // 2, 2))\n\n        q = self._apply_rope(q, freqs_cis)\n        k = self._apply_rope(k, freqs_cis)\n\n        # Flatten spatial dims for attention: (B, Nh, T*H*W, D)\n        N = T * H * W\n        q = q.reshape(B, N, Nh, D).permute(0, 2, 1, 3)\n        k = k.reshape(B, N, Nh, D).permute(0, 2, 1, 3)\n        v = v.reshape(B, N, Nh, D).permute(0, 2, 1, 3)\n\n        out = F.scaled_dot_product_attention(q, k, v)  # (B, Nh, N, D)\n        out = out.permute(0, 2, 1, 3).reshape(B, T, H, W, C)\n        return x + self.proj(out)\n\n\n# ---------------------------------------------------------------------------\n# Helpers\n# ---------------------------------------------------------------------------\n\n\ndef make_inputs(\n    B: int, T: int, H: int, W: int, C: int, n_heads: int\n) -> tuple[torch.Tensor, torch.Tensor]:\n    \"\"\"Return (x, freqs_cis_real) on DEVICE.\"\"\"\n    x = torch.randn(B, T, H, W, C, dtype=torch.float32, device=DEVICE)\n    freqs_cis = precompute_freqs_3d(C // n_heads, t=T, h=H, w=W).to(DEVICE)\n    freqs_cis_real = torch.view_as_real(freqs_cis).flatten(-2)  # (T,H,W,D)\n    return x, freqs_cis_real\n\n\ndef benchmark(fn, *args, iterations: int = 20, label: str = \"\") -> float:\n    fn(*args)  # warmup\n    torch.cuda.synchronize()\n    total = 0.0\n    for _ in range(iterations):\n        t0 = timeit.default_timer()\n        fn(*args)\n        torch.cuda.synchronize()\n        total += timeit.default_timer() - t0\n    avg_ms = total / iterations * 1000\n    print(f\"[{label}] avg latency over {iterations} iters: {avg_ms:.2f} ms\")\n    return avg_ms\n\n\n# ---------------------------------------------------------------------------\n# Main\n# ---------------------------------------------------------------------------\n\n\ndef parse_args():\n    p = argparse.ArgumentParser(\n        description=\"3D RoPE attention block compiled with Torch-TensorRT\"\n    )\n    p.add_argument(\"--heads\", type=int, default=8, help=\"Number of attention heads\")\n    p.add_argument(\n        \"--head-dim\",\n        dest=\"head_dim\",\n        type=int,\n        default=48,\n        help=\"Channels per head. head_dim//2 must be divisible by 3 (default: 48)\",\n    )\n    p.add_argument(\"--t\", type=int, default=4, help=\"Temporal frames (default: 4)\")\n    p.add_argument(\n        \"--h\", type=int, default=8, help=\"Spatial height patches (default: 8)\"\n    )\n    p.add_argument(\n        \"--w\", type=int, default=8, default=20)\n    return p.parse_args()\n\n\ndef main():\n    args = parse_args()\n\n    if (args.head_dim // 2) % 3 != 0:\n        raise ValueError(\n            f\"head_dim // 2 = {args.head_dim // 2} must be divisible by 3 \"\n            \"for the T/H/W frequency split.  Try --head-dim 48, 60, 96, or 192.\"\n        )\n\n    B, T, H, W = 1, args.t, args.h, args.w\n    C = args.heads * args.head_dim\n\n    print(f\"VideoAttentionBlock with 3D RoPE\")\n    print(f\"  heads={args.heads}  head_dim={args.head_dim}  channels={C}\")\n    print(f\"  input shape: ({B}, {T}, {H}, {W}, {C})\")\n\n    model = VideoAttentionBlock(channels=C, n_heads=args.heads).eval().to(DEVICE)\n\n    # ------------------------------------------------------------------\n    # 1. Build inputs\n    # ------------------------------------------------------------------\n    x, freqs_cis_real = make_inputs(B, T, H, W, C, args.heads)\n    inputs = (x, freqs_cis_real)\n    print(f\"\\n  x shape             : {x.shape}\")\n    print(f\"  freqs_cis_real shape: {freqs_cis_real.shape}\")\n\n    # ------------------------------------------------------------------\n    # 2. PyTorch baseline\n    # ------------------------------------------------------------------\n    with torch.inference_mode():\n        pyt_out = model(*inputs)\n    print(f\"\\n--- PyTorch baseline ---\")\n    print(f\"  output shape: {pyt_out.shape}  dtype: {pyt_out.dtype}\")\n\n    # ------------------------------------------------------------------\n    # 3. Export\n    # ------------------------------------------------------------------\n    print(\"\\nExporting model ...\")\n    if args.dynamic:\n        t_dim = Dim(\"T\", min=1, max=32)\n        h_dim = Dim(\"H\", min=4, max=64)\n        w_dim = Dim(\"W\", min=4, max=64)\n        dynamic_shapes = (\n            # x: (B, T, H, W, C)\n            {1: t_dim, 2: h_dim, 3: w_dim},\n            # freqs_cis_real: (T, H, W, D)\n            {0: t_dim, 1: h_dim, 2: w_dim},\n        )\n        ep = torch.export.export(model, inputs, dynamic_shapes=dynamic_shapes)\n        print(\"  Exported with dynamic T / H / W dimensions.\")\n    else:\n        ep = torch.export.export(model, inputs)\n        print(\"  Exported with static shapes.\")\n\n    # ------------------------------------------------------------------\n    # 4. Compile with Torch-TensorRT\n    #\n    # No special flags are required for the complex arithmetic rewrite.\n    # The complex_graph_detection lowering pass automatically detects\n    # view_as_complex / complex-mul / view_as_real subgraphs and rewrites\n    # them to real-arithmetic ops before the TRT engine is built.\n    # ------------------------------------------------------------------\n    print(\"\\nCompiling with Torch-TensorRT ...\")\n    D = C // args.heads  # freqs_cis_real last dim\n    if args.dynamic:\n        trt_inputs = [\n            torch_tensorrt.Input(\n                min_shape=(B, 1, 4, 4, C),\n                opt_shape=(B, T, H, W, C),\n                max_shape=(B, 32, 64, 64, C),\n                dtype=torch.float32,\n            ),\n            torch_tensorrt.Input(\n                min_shape=(1, 4, 4, D),\n                opt_shape=(T, H, W, D),\n                max_shape=(32, 64, 64, D),\n                dtype=torch.float32,\n            ),\n        ]\n    else:\n        trt_inputs = list(inputs)\n\n    trt_model = torch_tensorrt.dynamo.compile(\n        ep,\n        inputs=trt_inputs,\n        min_block_size=1,\n    )\n\n    # ------------------------------------------------------------------\n    # 5.
